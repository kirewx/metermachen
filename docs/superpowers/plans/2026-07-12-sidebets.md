```python
    reason: str  # "start" | "einkommen" | "einsatz" | "gewinn" | "rueckzahlung"

    created_at: datetime = Field(default_factory=utcnow)


class Bet(SQLModel, table=True):
    id: int | None = Field(default=None, primary_key=True)
    type: str  # "duell" | "monats_tipp" | "ziel" | "streak" | "ueber_unter"

    period_end: date_type

    jackpot: int = 0  # nur monats_tipp: Übertrag aus Vormonat
    created_at: datetime = Field(default_factory=utcnow)
    resolved_at: datetime | None = None
    result_json: str = "{}"  # nach Auflösung: Ist-Werte, Gewinner-IDs


class BetParticipant(SQLModel, table=True):
    id: int | None = Field(default=None, primary_key=True)
    bet_id: int = Field(foreign_key="bet.id", index=True)
    user_id: int = Field(foreign_key="user.id", index=True)
    role: str  # "ersteller" | "gegner" | "tipper" | "gegenhalter" | "ueber" | "unter"
    choice_json: str = "{}"  # tipper: {"tipp_user_id": 5}; sonst leer
    stake: int = 0
```

```python
"""Punkte-Ledger: Kontostand = Summe aller PointTransactions.

Einkommen (+1 P je 5 gewertete km ab Challenge-Start) wird idempotent
nachgebucht: Soll = floor(km/5), gebucht wird nur die Differenz zum
bereits gutgeschriebenen Einkommen. Kein Cron nötig.
"""
import math
from datetime import date as date_type

from sqlmodel import Session, func, select

from ..models import Activity, Category, PointTransaction, Season, User

START_CREDIT = 100
KM_PER_POINT = 5.0


def balance(session: Session, user_id: int) -> int:
    total = session.exec(
        select(func.coalesce(func.sum(PointTransaction.amount), 0)).where(
            PointTransaction.user_id == user_id
        )
    ).one()
    return int(total)


def ensure_start_credit(session: Session, user_id: int) -> None:
    exists = session.exec(
        select(PointTransaction).where(
            PointTransaction.user_id == user_id, PointTransaction.reason == "start"
        )
    ).first()
    if exists is None:
        session.add(PointTransaction(user_id=user_id, amount=START_CREDIT, reason="start"))
        session.commit()


def _challenge_start(session: Session) -> date_type | None:
    season = session.exec(
        select(Season).where(Season.year == date_type.today().year)
    ).first()
    return season.start_date if season else None


def scaled_km_since_start(session: Session, user_id: int) -> float:
    start = _challenge_start(session)
    if start is None or date_type.today() < start:
        return 0.0
    user = session.get(User, user_id)
    cats = {c.id: c for c in session.exec(select(Category)).all()}
    acts = session.exec(
        select(Activity).where(Activity.user_id == user_id, Activity.date >= start)
    ).all()
    return sum(a.distance_km * cats[a.category_id].factor * user.km_factor
               for a in acts if a.category_id in cats)


def ensure_income(session: Session, user_id: int) -> None:
    earned = math.floor(scaled_km_since_start(session, user_id) / KM_PER_POINT)
    booked = session.exec(
        select(func.coalesce(func.sum(PointTransaction.amount), 0)).where(
            PointTransaction.user_id == user_id,
            PointTransaction.reason == "einkommen",
        )
    ).one()
    if earned > int(booked):
        session.add(PointTransaction(
            user_id=user_id, amount=earned - int(booked), reason="einkommen"
        ))
        session.commit()


def refresh_user(session: Session, user_id: int) -> None:
    ensure_start_credit(session, user_id)
    ensure_income(session, user_id)
```

```python
from collections import defaultdict
from datetime import date as date_type
from datetime import timedelta

from sqlmodel import Session, select

from ..models import Activity, Category


def scaled_km(session: Session, user_id: int,
              start: date_type, end: date_type) -> float:
    cats = {c.id: c for c in session.exec(select(Category)).all()}
    acts = session.exec(select(Activity).where(
        Activity.user_id == user_id,
        Activity.date >= start, Activity.date <= end,
    )).all()
    return round(sum(a.distance_km * cats[a.category_id].factor
                     for a in acts if a.category_id in cats), 2)


def group_scaled_km(session: Session, user_ids: list[int],
                    start: date_type, end: date_type) -> float:
    return round(sum(scaled_km(session, uid, start, end) for uid in user_ids), 2)


def longest_streak(session: Session, user_id: int,
                   start: date_type, end: date_type) -> int:
    """Längste Serie von Tagen mit >= 1 km roh im Zeitraum."""
    per_day: dict[date_type, float] = defaultdict(float)
    for a in session.exec(select(Activity).where(
        Activity.user_id == user_id,
        Activity.date >= start, Activity.date <= end,
    )).all():
        per_day[a.date] += a.distance_km
    best = run = 0
    day = start
    while day <= end:
        run = run + 1 if per_day[day] >= 1.0 else 0
        best = max(best, run)
        day += timedelta(days=1)
    return best
```

Regeln (aus Spec):
- `period_start` > heute; Beitritt/Tipp/Dagegenhalten nur bis `period_start` (Monats-Tipp: bis zum 5. des Monats).
- Einsatz wird sofort als Transaktion `einsatz` (negativ) gebucht; Kontostand darf nicht unterschritten werden (vorher `points.refresh_user`, dann prüfen).
- Duell: Status `offen` → Gegner `accept` (bucht seinen Einsatz, Status `laufend`) oder `decline` (Status `abgelehnt`, Ersteller-Einsatz zurück als `rueckzahlung`).
- Ziel/Streak/Über-Unter: Status direkt `laufend`, weitere Teilnehmer bis `period_start`.
- Gegenhalter-Summe ≤ Ersteller-Einsatz (Ziel/Streak).
- `cancel`: nur Ersteller, nur vor `period_start`, alle Einsätze zurück, Status `abgebrochen`.
- Monats-Tipp: `ensure_monthly_tip(session)` erzeugt beim ersten Zugriff im Monat M (ab 2026-08) die Runde: `period_start` = 1. des Monats, `period_end` = letzter Tag, Tipp (role `tipper`, stake 10, choice `tipp_user_id`) bis zum 5. — Ausnahme von der `period_start`-Zukunftsregel.

### Task 5: Wett-Engine — Auflösung

**Files:**
- Modify: `backend/app/services/bets.py` (+ `resolve_due`)
- Test: `backend/tests/test_bets_resolution.py`

Auszahlungsregeln (payout = Gesamtgutschrift `gewinn` bzw. `rueckzahlung`):
- **duell:** km_creator×factor_creator vs. (km_opponent + vorsprung)×factor_opponent → Gewinner +2×stake. Gleichstand: beide +stake zurück.
- **ziel/streak:** Ist ≥ Ziel → Ersteller +(eigener Einsatz + Σ Gegenhalter-Einsätze). Sonst: jeder Gegenhalter +(eigener Einsatz + anteilig vom Ersteller-Einsatz, `floor(stake_i / Σ stakes * ersteller_stake)`, Rest an den frühesten Gegenhalter). Keine Gegenhalter → Ersteller +stake zurück, Status `abgebrochen`.
- **ueber_unter:** Gruppen-km (alle aktiven User) vs. target. Gewinnerseite: +eigener Einsatz + `floor(stake_i / Σ Gewinner-Stakes * Σ Verlierer-Stakes)`, Rest an frühesten Gewinner. Eine Seite leer → alle zurück. Exakt = Ziel zählt als „Über" erreicht (dokumentieren!).
- **monats_tipp:** Sieger = User mit meisten gewerteten km (MIT `km_factor`, wie Haupt-Ranking) im Monat. Pott = Σ Tipp-Einsätze + jackpot. Richtige Tipper teilen zu gleichen Teilen (`floor`), Rest wandert als `jackpot` in die Runde des Folgemonats (wird bei Bedarf sofort angelegt). Niemand richtig → gesamter Pott in Folgemonat. Duell offen & `period_start` erreicht ohne Annahme → `abgebrochen`, Ersteller zurück.
- Deaktivierter/gelöschter Teilnehmer → Wette `abgebrochen`, alle zurück.
- Jede Auflösung: `status="entschieden"`, `resolved_at`, `result_json` mit Ist-Werten (`{"km": {...}, "winner_ids": [...]}`), `payout` je Teilnehmer. Beim Duell zusätzlich in `result_json`: `{"david": true}` wenn der Gewinner beim Auflösen im KM-Jahresranking (Challenge-Phase) hinter dem Verlierer stand.

Endpoints (Auth wie Bestand über `get_current_user`):

```
GET  /api/bets                  -> ensure_monthly_tip + resolve_due, dann BetOut-Liste
                                   (inkl. Teilnehmer, Live-Zwischenstand für laufende Wetten)
POST /api/bets                  -> create_bet; ValueError -> HTTP 400
POST /api/bets/{id}/respond     -> respond (Body: {action, stake?, choice?})
POST /api/bets/{id}/cancel      -> cancel
GET  /api/points                -> refresh_user, dann {balance, transactions[]}
GET  /api/points/ranking        -> refresh alle aktiven User, dann [{user_id, display_name, avatar, balance}] sortiert
GET  /api/bets/achievements     -> Wett-Achievements (siehe unten) für alle User
```

`BetOut` enthält: alle Bet-Felder + `params` (geparst) + `participants` (mit display_name/avatar) + `standing` (typabhängiger Live-Stand, z.B. `{"creator_km": 12.3, "opponent_km": 8.1}`) + `my_role`.

Wett-Achievements (Berechnung live aus entschiedenen Wetten + `result_json`):
- `zocker`: ≥10 Wetten mit Status `entschieden`, an denen der User teilnahm
- `david`: Duell gewonnen mit `result_json.david == true`
- `high_roller`: Wette mit Einsatz ≥ 50 gewonnen (payout > stake)
- `orakel`: ≥3 Monats-Tipps mit richtigem Tipp

Response-Form wie `AchievementOut`-Liste pro User: `[{user_id, achievements: [{key, title, achieved, progress}]}]` — nur eigene + Übersichtszähler reicht: `GET /api/bets/achievements` liefert für den eingeloggten User.
